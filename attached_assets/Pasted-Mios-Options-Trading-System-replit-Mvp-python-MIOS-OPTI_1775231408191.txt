def generate_signal(data):
    regime = detect_market_regime(data['price'], data['ma50'], data['ma200'])
    strategy = select_strategy(regime, data['shares'], data['cash'])
    iv_state = iv_filter(data['iv'], data['iv_hist'])


    if iv_state == "avoid":
        return None


    if strategy == "covered_call":
        strike = select_call_strike(data['price'])
        return {"action": "SELL_CALL", "strike": strike}


    if strategy == "cash_secured_put":
        strike = select_put_strike(data['price'])
        return {"action": "SELL_PUT", "strike": strike}


    return None
