3. Database Manager
database/db_manager.py
import sqlite3

class DBManager:

    def __init__(self, db_path="investment_os.db"):
        self.conn = sqlite3.connect(db_path)
        self.conn.row_factory = sqlite3.Row

    def insert_dataframe(self, df, table):
        df.to_sql(table, self.conn, if_exists="append", index=False)

    def query(self, sql):
        cursor = self.conn.cursor()
        cursor.execute(sql)
        return cursor.fetchall()

    def get_company_history(self, ticker, limit=20):

        query = f"""
        SELECT *
        FROM financial_metrics
        WHERE ticker='{ticker}'
        ORDER BY date DESC
        LIMIT {limit}
        """

        return self.query(query)

import requests
import pandas as pd
from database.db_manager import DBManager

API_KEY = "YOUR_FMP_KEY"

class DataHarvester:

    def __init__(self):
        self.db = DBManager()

    def fetch_financials(self, ticker):

        url = f"https://financialmodelingprep.com/api/v3/key-metrics/{ticker}?apikey={API_KEY}"

        r = requests.get(url)
        data = r.json()

        df = pd.DataFrame(data)

        df = df.rename(columns={
            "roic": "roic",
            "debtToEquity": "debt_to_equity",
            "freeCashFlowYield": "fcf_yield"
        })

        df["ticker"] = ticker

        return df

    def store(self, df):

        self.db.insert_dataframe(df, "financial_metrics")

import numpy as np

def calculate_fortress_score(data):

    roic = np.mean(data["roic"])
    debt = data["debt_to_equity"].iloc[0]
    fcf = data["fcf_yield"].iloc[0]

    roic_score = roic * 0.4
    debt_score = (1 / (debt + 0.1)) * 0.3
    fcf_score = fcf * 0.3

    return roic_score + debt_score + fcf_score

import numpy as np

def calculate_rocket_score(data):

    rd_intensity = np.mean(data["rd_to_revenue"])
    insider = data["insider_ownership"].iloc[0]
    revenue_growth = np.mean(data["revenue"])

    score = (
        rd_intensity * 0.5 +
        insider * 0.3 +
        revenue_growth * 0.2
    )

    return score
Wave Engine (Tactical)
engines/wave_engine.py
from utils.indicators import moving_average, rsi

def calculate_wave_score(price_df):

    ma50 = moving_average(price_df["close"], 50)
    ma200 = moving_average(price_df["close"], 200)

    golden_cross = ma50.iloc[-1] > ma200.iloc[-1]

    rsi_value = rsi(price_df["close"]).iloc[-1]

    score = 0

    if golden_cross:
        score += 0.6

    if rsi_value < 70:
        score += 0.4

    return score
6. Indicators Library
utils/indicators.py
import pandas as pd

def moving_average(series, window):

    return series.rolling(window).mean()

def rsi(series, window=14):

    delta = series.diff()

    gain = (delta.where(delta > 0, 0)).rolling(window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window).mean()

    rs = gain / loss

    return 100 - (100 / (1 + rs))

4. Nightly Investment Pipeline

This runs automatically once per day.

pipeline/nightly_pipeline.py

from data_layer.universe_manager import UniverseManager
from data_layer.harvester import DataHarvester
from engines.fortress_engine import calculate_fortress_score
from engines.rocket_engine import calculate_rocket_score
from engines.wave_engine import calculate_wave_score

class NightlyPipeline:

    def __init__(self):

        self.universe = UniverseManager()
        self.harvester = DataHarvester()

    def run(self):

        tickers = self.universe.get_all_tickers()

        for ticker in tickers:

            try:

                print("Processing:", ticker)

                financials = self.harvester.fetch_financials(ticker)
                prices = self.harvester.fetch_prices(ticker)

                fortress = calculate_fortress_score(financials)
                rocket = calculate_rocket_score(financials)
                wave = calculate_wave_score(prices)

                self.harvester.store_scores(
                    ticker,
                    fortress,
                    rocket,
                    wave
                )

            except Exception as e:

                print("Failed:", ticker, e)

def detect_factor_drift(current, historical):

    signals = []

    if current["roic"] < historical["roic"].mean() * 0.8:
        signals.append("ROIC deterioration")

    if current["debt_to_equity"] > historical["debt_to_equity"].mean() * 1.5:
        signals.append("Debt rising")

    if current["receivables_growth"] > current["revenue_growth"]:
        signals.append("Possible channel stuffing")

    return signals
